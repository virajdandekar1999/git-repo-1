import boto3
import pandas as pd
from io import StringIO

# Configuration
bucket_name = "student-etl-viraj"
input_key = "sales_file.csv"
output_key = "output/filtered_file.csv"

# Initialize S3 client
s3 = boto3.client("s3")

# Step 1: Read CSV from S3
response = s3.get_object(Bucket=bucket_name, Key=input_key)
csv_content = response['Body'].read().decode('utf-8')

# Load into pandas DataFrame
df = pd.read_csv(StringIO(csv_content))

# Step 2: Filter rows where amount > 1000
filtered_df = df[df['amount'] > 1000]

# Step 3: Convert filtered DataFrame to CSV (in-memory)
output_buffer = StringIO()
filtered_df.to_csv(output_buffer, index=False)

# Step 4: Upload filtered CSV back to S3
s3.put_object(
    Bucket=bucket_name,
    Key=output_key,
    Body=output_buffer.getvalue()
)

print(f"Filtered file uploaded to s3://{bucket_name}/{output_key}")
